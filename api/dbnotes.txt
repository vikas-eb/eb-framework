
--Create the user table with following SQL

CREATE TABLE `boilerplate`.`user` ( `Id` VARCHAR(256) NOT NULL , `Email` VARCHAR(256) NOT NULL , `Password` VARCHAR(256) NOT NULL , `FirstName` VARCHAR(100) NOT NULL , `MiddleName` VARCHAR(100) NOT NULL , `LastName` VARCHAR(100) NOT NULL , `Phone` VARCHAR(20) NOT NULL , `Address` VARCHAR(1000) NOT NULL , `City` VARCHAR(50) NOT NULL , `State` VARCHAR(50) NOT NULL , `Zip` VARCHAR(10) NOT NULL , `DOB` DATE NOT NULL , `CreatedBy` VARCHAR(256) NOT NULL , `UpdatedBy` VARCHAR(256) NOT NULL , `CreatedAt` DATETIME NOT NULL , `UpdatedAt` DATETIME NOT NULL , PRIMARY KEY (`Id`), UNIQUE (`Email`)) ENGINE = InnoDB;
